Tables de la Base de Données (Projet)

Utilisateurs (users)
user_id : INT, PRIMARY KEY, AUTO_INCREMENT
username : VARCHAR, unique

Quiz (quizzes)
quiz_id : INT, PRIMARY KEY, AUTO_INCREMENT
user_id : INT, FOREIGN KEY référençant users.user_id
quiz_title : VARCHAR
created_at : DATETIME

Questions (questions)
question_id : INT, PRIMARY KEY, AUTO_INCREMENT
quiz_id : INT, FOREIGN KEY référençant quizzes.quiz_id
question_text : TEXT

Réponses (answers)
answer_id : INT, PRIMARY KEY, AUTO_INCREMENT
question_id : INT, FOREIGN KEY référençant questions.question_id
answer_text : TEXT
is_correct : BOOLEAN (TRUE pour la bonne réponse, FALSE pour les autres)
Réponses des Utilisateurs (user_answers)

user_answer_id : INT, PRIMARY KEY, AUTO_INCREMENT
question_id : INT, FOREIGN KEY référençant questions.question_id
answer_id : INT, FOREIGN KEY référençant answers.answer_id
username : VARCHAR
